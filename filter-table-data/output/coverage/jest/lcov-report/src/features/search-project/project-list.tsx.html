import React from "react";
import { IProjectList } from "./search-project";
 
interface IPropsProjectList {
    projectList: IProjectList[];
}
 
 
const ProjectsList = (props: IPropsProjectList) => {
    return <div className="">
            <table className="table">
                <thead>
                    <tr>
                        <th>Project Id</th>
                        <th>Project Name</th>
                        <th>Type</th>
                        <th>Technology</th>
                    </tr>
                </thead>
                <tbody>
                    {
                        props.projectList.map((item: IProjectList, index: number) => {
                            return <tr key={index}>
                                        <td>{item.pid}</td>
                                        <td>{item.pName}</td>
                                        <td>{item.type}</td>
                                        <td>{item.tech}</td>
                                    </tr>
                        })
                    }
                                            
                </tbody>
            </table>                
        </div>
}
 
export default ProjectsList;
