import React from 'react';
import { Form, Button } from 'react-bootstrap';
import ProjectsList from './project-list';
 
export interface IProjectList {
    pid: number;
    pName: string;
    type: string;
    tech: string;
}
interface IProject {
    searchedTxt: string;
    projectList: IProjectList[];
}
  
const defaultProjectList: IProjectList[] = [
    {
      pid: 101,
      pName: "abc",
      type: "Automation",
      tech: "Testing"
    },
    {
      pid: 102,
      pName: "abcd",
      type: "Manual",
      tech: "Testing"
    }
];
 
class SearchProject extends React.Component {
    state: IProject = {
        searchedTxt: "",
        projectList: defaultProjectList
    }
    render() {
        return (
            <div className="container-fluid">
                <div className="d-flex justify-content-center align-items-center">
                    <Form>
                        <Form.Row>
                            <Form.Group>
                                <Form.Control
                                    name="projectName"
                                    placeholder="Project name or ID" />                                    
                            </Form.Group>
                            <Form.Group>
                                <Button type="submit" variant="primary">Submit</Button>
                            </Form.Group>
                        </Form.Row>
                    </Form>
                </div>
                <div className="col-md-12">
                    <ProjectsList projectList={this.state.projectList}></ProjectsList>
                </div>                
            </div>
        )
    }
}
export default SearchProject
